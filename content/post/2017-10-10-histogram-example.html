library(Lahman)
library(ggplot2)
# extracting the data------------------

query<-"SELECT weight
FROM Master"

result<-sqldf(query)

# visualize the data-------------------

ggplot()+
  geom_histogram(data=result,aes(x=weight),color="blue",fill="white")+
  ggtitle("Baseball Player Body-Weight Distribution")
